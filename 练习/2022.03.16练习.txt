#include<iostream>

using namespace std;

int Fibonacci(int i)

{

if (i == 1 || i == 2) return 1;

return Fibonacci(i - 1) + Fibonacci(i - 2);

}

int main()

{

int* space = new int[20];

if (space == NULL)

{

return 0;

}

for (int i = 0; i < 20; ++i)

{

*(space+i) = Fibonacci(i + 1);

}

for (int i = 0; i < 20; ++i)

{

cout << *(space+i) << endl;

}

if (space != NULL)

{

delete[] space;

space = nullptr;

}

return 0;

}
